## Usage

```tsx
import { Checkbox } from "shadmin/components/ui/checkbox";
import { Label } from "shadmin/components/ui/label";

<div className="flex items-center gap-2">
  <Checkbox id="terms" />
  <Label htmlFor="terms">Accept terms and conditions</Label>
</div>
```
